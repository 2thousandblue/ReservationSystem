drop database if exists familydoctor;

create database familydoctor character set utf8;

use familydoctor;

create table doctor(
id int primary key auto_increment,
doctorName varchar(20),
sex varchar(10),
phoneNum varchar(15),
attend varchar(20),
professional varchar(30),
dateTime date)engine=innodb default charset=utf8;




姓名、年龄、性别、电话、住址、病史、编号
create table inhabitant(
id int primary key auto_increment,
inhabitantName varchar(20),
age int,
sex varchar(10),
phoneNum varchar(15),
address varchar(20),
anamnesis varchar(100)
)engine=innodb default charset=utf8;
